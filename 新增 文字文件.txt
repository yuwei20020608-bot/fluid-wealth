        // 3. iOS 橡皮筋滑動與長按手勢監聽
        function initSliderTouch() {
            const slider = document.getElementById("main-slider");
            let startX = 0, currentX = 0, isDragging = false;

            window.addEventListener("touchstart", (e) => {
                if(e.target.closest(".blur-overlay") || e.target.closest(".card-details")) return;
                startX = e.touches.clientX;
                isDragging = true;
                slider.style.transition = "none";
            });

            window.addEventListener("touchmove", (e) => {
                if(!isDragging) return;
                currentX = e.touches.clientX;
                let diff = currentX - startX;
                let baseTransform = -appState.currentTab * (window.innerWidth);
                slider.style.transform = `translateX(${baseTransform + diff}px)`;
            });

            window.addEventListener("touchend", (e) => {
                if(!isDragging) return;
                isDragging = false;
                slider.style.transition = "transform 0.4s cubic-bezier(0.25, 1, 0.5, 1)";
                let diff = currentX - startX;

                if(Math.abs(diff) > window.innerWidth / 5) {
                    if(diff > 0 && appState.currentTab > 0) appState.currentTab--;
                    else if(diff < 0 && appState.currentTab < 2) appState.currentTab++;
                }
                slider.style.transform = `translateX(${-appState.currentTab * window.innerWidth}px)`;
            });
        }

        function initRescueCarousel() {
            const overlay = document.getElementById("rescue-panel-overlay");
            let touchStartX = 0;

            overlay.addEventListener("touchstart", (e) => {
                touchStartX = e.touches.clientX;
            });

            overlay.addEventListener("touchend", (e) => {
                let diff = e.changedTouches.clientX - touchStartX;
                if(Math.abs(diff) > 60) {
                    if(diff < 0 && appState.activeRescueIdx < 3) appState.activeRescueIdx++;
                    if(diff > 0 && appState.activeRescueIdx > 0) appState.activeRescueIdx--;
                    switchRescueCard();
                }
            });
        }

        function switchRescueCard() {
            const cards = document.querySelectorAll("#rescue-cards-stack .rescue-card");
            const dots = document.querySelectorAll(".carousel-dots .dot");
            
            cards.forEach((card, i) => {
                card.className = "rescue-card";
                if(i === appState.activeRescueIdx) card.classList.add("active-center");
                else if(i < appState.activeRescueIdx) card.classList.add("hidden-left");
                else card.classList.add("hidden-right");
            });

            dots.forEach((dot, i) => {
                if(i === appState.activeRescueIdx) dot.classList.add("active-center", "active");
                else dot.className = "dot";
            });
        }

        function executeClientAccounting() {
            const acc = document.getElementById("form-log-account").value;
            const type = document.getElementById("form-log-type").value;
            const amt = parseFloat(document.getElementById("form-log-amount").value) || 0;
            const label = document.getElementById("form-log-label").value;
            const proj = document.getElementById("form-log-project").value;

            if(amt <= 0) return alert("請輸入正確金額");

            if(acc === "中國信託" && type.indexOf("外輸出") > -1) {
                let currentMonthSpent = 0;
                appState.logs.forEach(log => {
                    if(log["帳戶"] === "中國信託" && log["流向分類"] && log["流向分類"].indexOf("外輸出") > -1) {
                        currentMonthSpent += parseFloat(log["金額"]) || 0;
                    }
                });
                
                if(currentMonthSpent + amt > appState.monthlyBudgetLimit) {
                    document.getElementById("overlay-add-log").classList.remove("active");
                    
                    const tbbCurrent = appState.balances["台灣企銀"] || 0;
                    const rescueTbbBtn = document.getElementById("btn-execute-rescue-tbb");
                    const rescueTbbStatus = document.getElementById("txt-rescue-tbb-status");
                    if(tbbCurrent < 3000) {
                        rescueTbbBtn.disabled = true;
                        rescueTbbBtn.style.background = "#333";
                        rescueTbbStatus.innerText = "[🚨 警告：台企銀低於 $3,000 死線已鎖死]";
                        rescueTbbStatus.style.color = "var(--neon-red)";
                    }
                    
                    document.getElementById("rescue-panel-overlay").classList.add("active");
                    return;
                }
            }

            sendAccountingToCloud(acc, type, amt, label, proj);
        }

        function sendAccountingToCloud(account, type, amount, label, project) {
            document.getElementById("overlay-add-log").classList.remove("active");
            fetch(`${API_URL}?action=addLog&account=${encodeURIComponent(account)}&type=${encodeURIComponent(type)}&amount=${amount}&label=${encodeURIComponent(label)}&project=${encodeURIComponent(project)}`)
                .then(res => res.json())
                .then(res => {
                    if(res.status === "success") {
                        refreshCloudData();
                    }
                });
        }

        function executeCreateProject() {
            const pName = document.getElementById("form-p-name").value;
            const pType = document.getElementById("form-p-type").value;
            const target = parseFloat(document.getElementById("form-p-target").value) || 0;

            if(!pName || target <= 0) return alert("請輸入完整計畫資料欄位");

            document.getElementById("overlay-settings").classList.remove("active");
            fetch(`${API_URL}?action=addProject&projectName=${encodeURIComponent(pName)}&projectType=${encodeURIComponent(pType)}&targetAmount=${target}&account=${pType==="儲蓄計畫"?"台中銀行":"中國信託"}&label=旅遊`)
                .then(res => res.json())
                .then(res => {
                    if(res.status === "success") {
                        refreshCloudData();
                    }
                });
        }

        function executeDirectRescue(rescueSource) {

        function executeLoanRescueAction() {
            const lender = "下個月的自己";
            const amt = parseFloat(document.getElementById("input-loan-amount").value) || 0;
            if(amt <= 0) return alert("請確認透支借款金額");

            fetch(`${API_URL}?action=addDebt&lender=${encodeURIComponent(lender)}&amount=${amt}`)
                .then(res => res.json())
                .then(res => {
                    if(res.status === "success") {
                        document.getElementById("rescue-panel-overlay").classList.remove("active");
                        document.getElementById("rescue-cards-stack").classList.remove("back-view");
                        refreshCloudData();
                    }
                });
        }
